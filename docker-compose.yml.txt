version: '3.8'

services:
  db:
    image: postgres:15
    container_name: postgres_db
    environment:
      POSTGRES_USER: devuser
      POSTGRES_PASSWORD: devpass
      POSTGRES_DB: devdb
    ports:
      - "5432:5432"
    volumes:
      - pgdata:/var/lib/postgresql/data

  backend:
    build: ./server/MyApp
    container_name: backend_api
    ports:
      - "5000:80"
    environment:
      ConnectionStrings__DefaultConnection: Host=db;Database=devdb;Username=devuser;Password=devpass
    depends_on:
      - db

  pgdata:
